import { useState, useEffect } from "react";
import { MapPin, Hospital, Droplet, Ambulance, Shield, List, X } from "lucide-react";
import Layout from "@/components/Layout";

interface MapResource {
  id: string;
  name: string;
  category: "hospitals" | "blood-banks" | "ambulance" | "police";
  address: string;
  phone: string;
  distance: number;
  latitude: number;
  longitude: number;
}

const MOCK_MAP_RESOURCES: MapResource[] = [
  {
    id: "1",
    name: "City General Hospital",
    category: "hospitals",
    address: "123 Healthcare Ave, Medical District",
    phone: "+1-555-0101",
    distance: 0.8,
    latitude: 40.7128,
    longitude: -74.006,
  },
  {
    id: "2",
    name: "St. Mary's Medical Center",
    category: "hospitals",
    address: "456 Medical Plaza, Downtown",
    phone: "+1-555-0102",
    distance: 1.2,
    latitude: 40.715,
    longitude: -74.007,
  },
  {
    id: "3",
    name: "Central Blood Bank",
    category: "blood-banks",
    address: "789 Donor Lane, East Side",
    phone: "+1-555-0201",
    distance: 1.5,
    latitude: 40.718,
    longitude: -74.005,
  },
  {
    id: "5",
    name: "Emergency Ambulance Service",
    category: "ambulance",
    address: "555 Response Ave, Central",
    phone: "+1-555-0301",
    distance: 0.5,
    latitude: 40.714,
    longitude: -74.008,
  },
  {
    id: "7",
    name: "Downtown Police Station",
    category: "police",
    address: "999 Law Enforcement Plaza, Downtown",
    phone: "+1-555-0401",
    distance: 1.1,
    latitude: 40.7135,
    longitude: -74.007,
  },
];

const CATEGORY_ICONS: Record<string, React.ReactNode> = {
  hospitals: "🏥",
  "blood-banks": "🩸",
  ambulance: "🚑",
  police: "👮",
};

const CATEGORY_COLORS: Record<string, string> = {
  hospitals: "bg-red-50 dark:bg-red-900/20 border-red-200 dark:border-red-900/50",
  "blood-banks": "bg-blue-50 dark:bg-blue-900/20 border-blue-200 dark:border-blue-900/50",
  ambulance: "bg-yellow-50 dark:bg-yellow-900/20 border-yellow-200 dark:border-yellow-900/50",
  police: "bg-purple-50 dark:bg-purple-900/20 border-purple-200 dark:border-purple-900/50",
};

export default function Map() {
  const [selectedCategory, setSelectedCategory] = useState<string | null>(null);
  const [selectedResource, setSelectedResource] = useState<MapResource | null>(null);
  const [userLocation, setUserLocation] = useState<{ lat: number; lng: number } | null>(null);
  const [showSidebar, setShowSidebar] = useState(true);

  const filteredResources = selectedCategory
    ? MOCK_MAP_RESOURCES.filter((r) => r.category === selectedCategory)
    : MOCK_MAP_RESOURCES;

  useEffect(() => {
    // Get user location
    if ("geolocation" in navigator) {
      navigator.geolocation.getCurrentPosition(
        (position) => {
          setUserLocation({
            lat: position.coords.latitude,
            lng: position.coords.longitude,
          });
        },
        () => {
          // Default to NYC if geolocation fails
          setUserLocation({ lat: 40.7128, lng: -74.006 });
        }
      );
    }
  }, []);

  return (
    <Layout>
      <div className="h-[calc(100vh-8rem)] flex overflow-hidden">
        {/* Map Area */}
        <div className="flex-1 bg-slate-100 dark:bg-slate-900 relative">
          {/* Map Placeholder */}
          <div className="w-full h-full flex items-center justify-center bg-gradient-to-br from-blue-100 to-blue-50 dark:from-blue-900/30 dark:to-blue-900/10">
            <div className="text-center space-y-4 pointer-events-none">
              <MapPin className="w-16 h-16 text-primary mx-auto opacity-50" />
              <div>
                <h3 className="text-xl font-bold text-foreground mb-2">Interactive Map</h3>
                <p className="text-muted-foreground max-w-md">
                  Resources are displayed as markers on the map. Click on resources in the sidebar to view details.
                </p>
              </div>
              {userLocation && (
                <p className="text-sm text-muted-foreground">
                  Your location: {userLocation.lat.toFixed(4)}, {userLocation.lng.toFixed(4)}
                </p>
              )}
              <div className="mt-6 space-y-2 text-sm text-muted-foreground">
                <p>📍 Integration ready with Leaflet, Mapbox, or Google Maps</p>
                <p>🎯 Support for custom markers and clustering</p>
                <p>📱 Mobile-friendly and responsive</p>
              </div>
            </div>
          </div>

          {/* Toggle Sidebar Button */}
          <button
            onClick={() => setShowSidebar(!showSidebar)}
            className="absolute bottom-6 left-6 z-20 p-3 bg-white dark:bg-card rounded-lg shadow-lg hover:shadow-xl transition-all border border-border"
          >
            <List className="w-5 h-5 text-foreground" />
          </button>
        </div>

        {/* Sidebar with Resources */}
        {showSidebar && (
          <div className="w-96 max-w-[100vw] bg-white dark:bg-card border-l border-border flex flex-col overflow-hidden">
            {/* Header */}
            <div className="bg-gradient-to-r from-slate-50 to-slate-100 dark:from-slate-900 dark:to-slate-800 p-4 border-b border-border">
              <div className="flex justify-between items-center mb-4">
                <h2 className="text-xl font-bold text-foreground">Resources</h2>
                <button
                  onClick={() => setShowSidebar(false)}
                  className="p-1 hover:bg-muted rounded-lg transition-colors"
                >
                  <X className="w-5 h-5 text-foreground" />
                </button>
              </div>

              {/* Category Filter */}
              <div className="space-y-2">
                <button
                  onClick={() => setSelectedCategory(null)}
                  className={`w-full text-left px-3 py-2 rounded-lg text-sm font-medium transition-all ${
                    selectedCategory === null
                      ? "bg-primary text-white"
                      : "bg-muted text-foreground hover:bg-muted/80"
                  }`}
                >
                  All Services
                </button>
                <div className="space-y-2">
                  {[
                    { id: "hospitals", label: "🏥 Hospitals" },
                    { id: "blood-banks", label: "🩸 Blood Banks" },
                    { id: "ambulance", label: "🚑 Ambulances" },
                    { id: "police", label: "👮 Police" },
                  ].map((cat) => (
                    <button
                      key={cat.id}
                      onClick={() => setSelectedCategory(cat.id)}
                      className={`w-full text-left px-3 py-2 rounded-lg text-sm font-medium transition-all ${
                        selectedCategory === cat.id
                          ? "bg-primary text-white"
                          : "bg-muted text-foreground hover:bg-muted/80"
                      }`}
                    >
                      {cat.label}
                    </button>
                  ))}
                </div>
              </div>
            </div>

            {/* Resources List */}
            <div className="flex-1 overflow-y-auto">
              <div className="p-4 space-y-3">
                {filteredResources.length > 0 ? (
                  filteredResources.map((resource) => (
                    <button
                      key={resource.id}
                      onClick={() => setSelectedResource(resource)}
                      className={`w-full text-left p-4 rounded-lg border-2 transition-all ${
                        selectedResource?.id === resource.id
                          ? `border-primary ${CATEGORY_COLORS[resource.category]}`
                          : `border-border hover:border-primary/30`
                      }`}
                    >
                      <div className="flex items-start gap-3">
                        <span className="text-2xl flex-shrink-0">
                          {CATEGORY_ICONS[resource.category]}
                        </span>
                        <div className="flex-1 min-w-0">
                          <h3 className="font-bold text-foreground truncate">
                            {resource.name}
                          </h3>
                          <p className="text-xs text-muted-foreground mt-1 line-clamp-2">
                            {resource.address}
                          </p>
                          <div className="flex items-center justify-between mt-2">
                            <a
                              href={`tel:${resource.phone}`}
                              className="text-xs text-primary hover:underline no-underline"
                              onClick={(e) => e.stopPropagation()}
                            >
                              {resource.phone}
                            </a>
                            <span className="text-xs font-bold text-primary">
                              {resource.distance} km
                            </span>
                          </div>
                        </div>
                      </div>
                    </button>
                  ))
                ) : (
                  <div className="text-center py-8">
                    <p className="text-muted-foreground">No resources found in this category</p>
                  </div>
                )}
              </div>
            </div>

            {/* Selected Resource Details */}
            {selectedResource && (
              <div className="border-t border-border p-4 bg-muted/50">
                <div className="space-y-3">
                  <h3 className="font-bold text-foreground">{selectedResource.name}</h3>
                  <div className="space-y-2 text-sm">
                    <p className="flex items-center gap-2 text-foreground">
                      <MapPin className="w-4 h-4 text-primary flex-shrink-0" />
                      <span className="text-muted-foreground">{selectedResource.address}</span>
                    </p>
                    <a
                      href={`tel:${selectedResource.phone}`}
                      className="flex items-center gap-2 text-primary hover:underline no-underline"
                    >
                      <Hospital className="w-4 h-4 flex-shrink-0" />
                      {selectedResource.phone}
                    </a>
                  </div>
                  <a
                    href={`tel:${selectedResource.phone}`}
                    className="block w-full px-4 py-2 bg-primary text-white rounded-lg font-medium text-center hover:bg-red-700 transition-all no-underline"
                  >
                    Call Now
                  </a>
                </div>
              </div>
            )}
          </div>
        )}
      </div>
    </Layout>
  );
}
